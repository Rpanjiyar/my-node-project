pipeline {
  agent any

  environment {
    SONAR_TOKEN = credentials('sonar-token')
  }

  tools {
    nodejs "NodeJS"
  }

  stages {
    stage('Checkout') {
      steps {
        git 'https://github.com/Rpanjiyar/my-node-project.git'
      }
    }

    stage('Install') {
      steps {
        bat 'npm install'
      }
    }

    stage('Test') {
      steps {
        bat 'npm test'
      }
    }

    stage('SonarQube Analysis') {
      steps {
        withSonarQubeEnv('MySonarQube') {
          bat 'npx sonar-scanner'
        }
      }
    }

    stage('Deploy to Local Folder') {
      steps {
        bat '''
        echo Deploying to local test env...
        if not exist "C:\\TestEnv\\my-node-project" mkdir "C:\\TestEnv\\my-node-project"
        xcopy /s /y * "C:\\TestEnv\\my-node-project"
        '''
      }
    }
  }
}
